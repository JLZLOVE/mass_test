CREATE TABLE club_application (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  application_no VARCHAR(50) NOT NULL COMMENT '申请编号',
  apply_type TINYINT NOT NULL COMMENT '1:创建 2:解散',
  
  -- 通用字段
  club_name VARCHAR(100) COMMENT '社团名称（创建时必填）',
  college_id BIGINT COMMENT '挂靠学院',
  category VARCHAR(50) COMMENT '社团类别',
  description TEXT COMMENT '社团简介',
  
  -- 创建申请专用
  proposed_leader_id BIGINT COMMENT '拟定社长ID',
  max_members INT COMMENT '最大招募人数',
  
  -- 解散申请专用
  dissolve_reason TEXT COMMENT '解散原因',
  
  -- 申请人
  applicant_id BIGINT NOT NULL COMMENT '申请人ID（指导老师）',
  applicant_name VARCHAR(50),
  
  -- 审批状态
  status TINYINT DEFAULT 1 COMMENT '1:待学院审批 2:学院已通过 3:已通过 4:已驳回 5:已撤回',
  reject_reason VARCHAR(500),
  
  -- 学院级审批
  college_approver_id BIGINT,
  college_approve_time DATETIME,
  college_approve_opinion VARCHAR(500),
  
  -- 校级审批
  admin_approver_id BIGINT,
  admin_approve_time DATETIME,
  admin_approve_opinion VARCHAR(500),
  
  create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
  update_time DATETIME ON UPDATE CURRENT_TIMESTAMP,
  
  UNIQUE INDEX uk_apply_no (application_no),
  INDEX idx_applicant (applicant_id),
  INDEX idx_college (college_id),
  INDEX idx_status (status)
);CREATE TABLE club_council (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  club_id BIGINT NOT NULL COMMENT '待解散社团ID',
  initiator_id BIGINT NOT NULL COMMENT '发起人ID（超管）',
  initiator_name VARCHAR(50),
  college_id BIGINT NOT NULL COMMENT '社团挂靠学院',
  reason TEXT NOT NULL COMMENT '解散原因',
  
  status TINYINT DEFAULT 1 COMMENT '1:合议中 2:已通过 3:已驳回',
  signatories JSON COMMENT '签名人ID列表（含角色信息）',
  
  executed_at DATETIME COMMENT '执行时间',
  create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
  
  INDEX idx_club (club_id),
  INDEX idx_college (college_id),
  INDEX idx_status (status)
);
